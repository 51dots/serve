import fs from "fs";
import { NextApiRequest, NextApiResponse } from "next";

export default function writeFile(req: NextApiRequest, res: NextApiResponse) {
  // File writing logic here
  const data = { contractAddress: "address" };
  fs.writeFile("test.json", JSON.stringify(data), (err) => {
    if (err) {
      console.error("Error writing to test.json:", err);
      res.status(500).json({ error: "Internal server error" });
    } else {
      console.log("test.json created successfully");
      res.status(200).json({ message: "File created successfully" });
    }
  });
}
